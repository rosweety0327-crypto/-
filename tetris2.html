class TetrisGame { constructor() { this.reset(); } reset() { this.grid = Array.from({length: ROWS}, () => Array(COLS).fill(0)); this.score = 0; scoreElement.innerText = this.score; // 重置顯示 this.spawn(); } spawn() { const id = Math.floor(Math.random() * 7); this.piece = { pos: {x: 3, y: 0}, shape: SHAPES[id], color: '#f1c40f' }; if (this.collide()) this.reset(); } collide() { const {pos, shape} = this.piece; for (let y = 0; y < shape.length; y++) { for (let x = 0; x < shape[y].length; x++) { if (shape[y][x] && (this.grid[y + pos.y]?.[x + pos.x] !== 0)) return true; } } return false; } rotate() { const s = this.piece.shape; const r = s[0].map((_, i) => s.map(row => row[i]).reverse()); const prev = this.piece.shape; this.piece.shape = r; if (this.collide()) this.piece.shape = prev; } move(dx, dy) { this.piece.pos.x += dx; this.piece.pos.y += dy; if (this.collide()) { this.piece.pos.x -= dx; this.piece.pos.y -= dy; if (dy > 0) this.freeze(); return false; } return true; } freeze() { this.piece.shape.forEach((row, y) => row.forEach((v, x) => { if (v) this.grid[y + this.piece.pos.y][x + this.piece.pos.x] = '#58a6ff'; })); this.clearLines(); // 這裡會檢查有沒有消行並給分 this.spawn(); } // 🎯 核心修改：只有消行才給分 clearLines() { let linesCleared = 0; // 檢查每一行是否全滿 this.grid = this.grid.filter(row => { const isFull = row.every(cell => cell !== 0); if (isFull) linesCleared++; // 記錄消了幾行 return !isFull; }); // 補回空行 while (this.grid.length < ROWS) { this.grid.unshift(Array(COLS).fill(0)); } // 只有在 linesCleared > 0 時才加分 if (linesCleared > 0) { // 基礎得分：1行100, 2行300, 3行500, 4行800 (獎勵連消) const points = [0, 100, 300, 500, 800]; this.score += points[linesCleared] || (linesCleared * 100); scoreElement.innerText = this.score; } } draw() { ctx.clearRect(0, 0, canvas.width, canvas.height); this.grid.forEach((row, y) => row.forEach((c, x) => { if (c) { ctx.fillStyle = c; ctx.fillRect(x*25, y*25, 24, 24); } })); ctx.fillStyle = this.piece.color; this.piece.shape.forEach((row, y) => row.forEach((v, x) => { if (v) ctx.fillRect((this.piece.pos.x+x)*25, (this.piece.pos.y+y)*25, 24, 24); })); } }
